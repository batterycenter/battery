325  if constexpr (Opts.force_conformance) {

335  case '\t': {
336  ctx.error = error_code::syntax_error;
337  return;
338  }
339  case '\\': {
340  ++it;
341  switch (*it) {
342  case '"':
343  case '\\':
344  case '/':
345  case 'b':
346  case 'f':
347  case 'n':
348  case 'r':
349  case 't': {
350  ++it;
351  continue;
352  }
353  case 'u': {
354  ++it;
355  if ((end - it) < 4) [[unlikely]] {
356  ctx.error = error_code::syntax_error;
357  return;
358  }
359  else if (std::all_of(it, it + 4, ::isxdigit)) [[likely]] {
360  it += 4;
361  }
362  else [[unlikely]] {
363  ctx.error = error_code::syntax_error;
364  return;
365  }
366  continue;
367  }
368  [[unlikely]] default:
369  {
370  ctx.error = error_code::syntax_error;
371  return;
372  }
373  }
374  }
375  [[likely]] default : ++it;
376  }
377  }
378  }
379  else {
380  while (it < end) {
381  if (*it == '"') {
382  ++it;
383  break;
384  }
385  else if (*it == '\\' && ++it == end) [[unlikely]]
386  break;
387  ++it;

471  GLZ_ALWAYS_INLINE void skip_number_with_validation(is_context auto&& ctx, auto&& it, auto&& end) noexcept
472  {
473  it += *it == '-';
474  const auto sig_start_it = it;
475  auto frac_start_it = end;
476  if (*it == '0') {
477  ++it;
478  if (*it != '.') {
479  return;
480  }
481  ++it;
482  goto frac_start;
483  }
484  it = std::find_if_not(it, end, is_digit);
485  if (it == sig_start_it) {
486  ctx.error = error_code::syntax_error;
487  return;
488  }
489  if ((*it | ('E' ^ 'e')) == 'e') {
490  ++it;
491  goto exp_start;
492  }
493  if (*it != '.') return;
494  ++it;
495  frac_start:
496  frac_start_it = it;
497  it = std::find_if_not(it, end, is_digit);
498  if (it == frac_start_it) {
499  ctx.error = error_code::syntax_error;
500  return;
501  }
502  if ((*it | ('E' ^ 'e')) != 'e') return;
503  ++it;
504  exp_start:
505  it += *it == '+' || *it == '-';
506  const auto exp_start_it = it;
507  it = std::find_if_not(it, end, is_digit);
508  if (it == exp_start_it) {
509  ctx.error = error_code::syntax_error;
510  return;
511  }
512  }
