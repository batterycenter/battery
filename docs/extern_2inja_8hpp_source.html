292  start = std::min(start, view.size());
293  end = std::min(std::max(start, end), view.size());

297  inline std::pair<std::string_view, std::string_view> split(std::string_view view, char Separator) {
298  size_t idx = view.find(Separator);
299  if (idx == std::string_view::npos) {
300  return std::make_pair(view, std::string_view());
301  }
302  return std::make_pair(slice(view, 0, idx), slice(view, idx + 1, std::string_view::npos));
303  }

310  inline SourceLocation get_source_location(std::string_view content, size_t pos) {
311  // Get line and offset position (starts at 1:1)
312  auto sliced = string_view::slice(content, 0, pos);
313  std::size_t last_newline = sliced.rfind("\n");
314 
315  if (last_newline == std::string_view::npos) {
316  return {1, sliced.length() + 1};
317  }
318 
319  // Count newlines
320  size_t count_lines = 0;
321  size_t search_start = 0;
322  while (search_start <= sliced.size()) {
323  search_start = sliced.find("\n", search_start) + 1;
324  if (search_start == 0) {
325  break;
326  }
327  count_lines += 1;
328  }
329 
330  return {count_lines + 1, sliced.length() - last_newline};
331  }

334  if (f.empty()) {
335  return;
336  }
337  for (auto pos = s.find(f); // find first occurrence of f

339  s.replace(pos, f.size(), t), // replace with t, and
340  pos = s.find(f, pos + t.size())) // find next occurrence of f
341  {}

1031  Token scan_body(std::string_view close, Token::Kind closeKind, std::string_view close_trim = std::string_view(), bool trim = false) {
1032  again:
1033  // skip whitespace (except for \n as it might be a close)
1034  if (tok_start >= m_in.size()) {
1035  return make_token(Token::Kind::Eof);
1036  }
1037  const char ch = m_in[tok_start];
1038  if (ch == ' ' || ch == '\t' || ch == '\r') {
1039  tok_start += 1;
1040  goto again;
1041  }
1042 
1043  // check for close
1044  if (!close_trim.empty() && inja::string_view::starts_with(m_in.substr(tok_start), close_trim)) {
1045  state = State::Text;
1046  pos = tok_start + close_trim.size();
1047  const Token tok = make_token(closeKind);
1048  skip_whitespaces_and_newlines();
1049  return tok;
1050  }
1051 
1052  if (inja::string_view::starts_with(m_in.substr(tok_start), close)) {
1053  state = State::Text;
1054  pos = tok_start + close.size();
1055  const Token tok = make_token(closeKind);
1056  if (trim) {
1057  skip_whitespaces_and_first_newline();
1058  }
1059  return tok;
1060  }
1061 
1062  // skip \n
1063  if (ch == '\n') {
1064  tok_start += 1;
1065  goto again;
1066  }
1067 
1068  pos = tok_start + 1;
1069  if (std::isalpha(ch)) {
1070  minus_state = MinusState::Operator;
1071  return scan_id();
1072  }
1073 
1074  const MinusState current_minus_state = minus_state;
1075  if (minus_state == MinusState::Operator) {
1076  minus_state = MinusState::Number;
1077  }
1078 
1079  switch (ch) {
1080  case '+':
1081  return make_token(Token::Kind::Plus);
1082  case '-':
1083  if (current_minus_state == MinusState::Operator) {
1084  return make_token(Token::Kind::Minus);
1085  }
1086  return scan_number();
1087  case '*':
1088  return make_token(Token::Kind::Times);
1089  case '/':
1090  return make_token(Token::Kind::Slash);
1091  case '^':
1092  return make_token(Token::Kind::Power);
1093  case '%':
1094  return make_token(Token::Kind::Percent);
1095  case '.':
1096  return make_token(Token::Kind::Dot);
1097  case ',':
1098  return make_token(Token::Kind::Comma);
1099  case ':':
1100  return make_token(Token::Kind::Colon);
1101  case '(':
1102  return make_token(Token::Kind::LeftParen);
1103  case ')':
1104  minus_state = MinusState::Operator;
1105  return make_token(Token::Kind::RightParen);
1106  case '[':
1107  return make_token(Token::Kind::LeftBracket);
1108  case ']':
1109  minus_state = MinusState::Operator;
1110  return make_token(Token::Kind::RightBracket);
1111  case '{':
1112  return make_token(Token::Kind::LeftBrace);
1113  case '}':
1114  minus_state = MinusState::Operator;
1115  return make_token(Token::Kind::RightBrace);
1116  case '>':
1117  if (pos < m_in.size() && m_in[pos] == '=') {
1118  pos += 1;
1119  return make_token(Token::Kind::GreaterEqual);
1120  }
1121  return make_token(Token::Kind::GreaterThan);
1122  case '<':
1123  if (pos < m_in.size() && m_in[pos] == '=') {
1124  pos += 1;
1125  return make_token(Token::Kind::LessEqual);
1126  }
1127  return make_token(Token::Kind::LessThan);
1128  case '=':
1129  if (pos < m_in.size() && m_in[pos] == '=') {
1130  pos += 1;
1131  return make_token(Token::Kind::Equal);
1132  }
1133  return make_token(Token::Kind::Unknown);
1134  case '!':
1135  if (pos < m_in.size() && m_in[pos] == '=') {
1136  pos += 1;
1137  return make_token(Token::Kind::NotEqual);
1138  }
1139  return make_token(Token::Kind::Unknown);
1140  case '\"':
1141  return scan_string();
1142  case '0':
1143  case '1':
1144  case '2':
1145  case '3':
1146  case '4':
1147  case '5':
1148  case '6':
1149  case '7':
1150  case '8':
1151  case '9':
1152  minus_state = MinusState::Operator;
1153  return scan_number();
1154  case '_':
1155  case '@':
1156  case '$':
1157  minus_state = MinusState::Operator;
1158  return scan_id();
1159  default:
1160  return make_token(Token::Kind::Unknown);
1161  }
1162  }

1165  for (;;) {
1166  if (pos >= m_in.size()) {
1167  break;
1168  }

1186  break;
1187  }
1188  pos += 1;
1189  }
1190  return make_token(Token::Kind::Number);

1265  void start(std::string_view input) {

1278  Token scan() {
1279  tok_start = pos;
1280 
1281  again:
1282  if (tok_start >= m_in.size()) {
1283  return make_token(Token::Kind::Eof);
1284  }
1285 
1286  switch (state) {
1287  default:
1288  case State::Text: {
1289  // fast-scan to first open character
1290  const size_t open_start = m_in.substr(pos).find_first_of(config.open_chars);
1291  if (open_start == std::string_view::npos) {
1292  // didn't find open, return remaining text as text token
1293  pos = m_in.size();
1294  return make_token(Token::Kind::Text);
1295  }
1296  pos += open_start;
1297 
1298  // try to match one of the opening sequences, and get the close
1299  std::string_view open_str = m_in.substr(pos);
1300  bool must_lstrip = false;
1301  if (inja::string_view::starts_with(open_str, config.expression_open)) {
1302  if (inja::string_view::starts_with(open_str, config.expression_open_force_lstrip)) {
1303  state = State::ExpressionStartForceLstrip;
1304  must_lstrip = true;
1305  } else {
1306  state = State::ExpressionStart;
1307  }
1308  } else if (inja::string_view::starts_with(open_str, config.statement_open)) {
1309  if (inja::string_view::starts_with(open_str, config.statement_open_no_lstrip)) {
1310  state = State::StatementStartNoLstrip;
1311  } else if (inja::string_view::starts_with(open_str, config.statement_open_force_lstrip)) {
1312  state = State::StatementStartForceLstrip;
1313  must_lstrip = true;
1314  } else {
1315  state = State::StatementStart;
1316  must_lstrip = config.lstrip_blocks;
1317  }
1318  } else if (inja::string_view::starts_with(open_str, config.comment_open)) {
1319  if (inja::string_view::starts_with(open_str, config.comment_open_force_lstrip)) {
1320  state = State::CommentStartForceLstrip;
1321  must_lstrip = true;
1322  } else {
1323  state = State::CommentStart;
1324  must_lstrip = config.lstrip_blocks;
1325  }
1326  } else if ((pos == 0 || m_in[pos - 1] == '\n') && inja::string_view::starts_with(open_str, config.line_statement)) {
1327  state = State::LineStart;
1328  } else {
1329  pos += 1; // wasn't actually an opening sequence
1330  goto again;
1331  }
1332 
1333  std::string_view text = string_view::slice(m_in, tok_start, pos);
1334  if (must_lstrip) {
1335  text = clear_final_line_if_whitespace(text);
1336  }
1337 
1338  if (text.empty()) {
1339  goto again; // don't generate empty token
1340  }
1341  return Token(Token::Kind::Text, text);
1342  }
1343  case State::ExpressionStart: {
1344  state = State::ExpressionBody;
1345  pos += config.expression_open.size();
1346  return make_token(Token::Kind::ExpressionOpen);
1347  }
1348  case State::ExpressionStartForceLstrip: {
1349  state = State::ExpressionBody;
1350  pos += config.expression_open_force_lstrip.size();
1351  return make_token(Token::Kind::ExpressionOpen);
1352  }
1353  case State::LineStart: {
1354  state = State::LineBody;
1355  pos += config.line_statement.size();
1356  return make_token(Token::Kind::LineStatementOpen);
1357  }
1358  case State::StatementStart: {
1359  state = State::StatementBody;
1360  pos += config.statement_open.size();
1361  return make_token(Token::Kind::StatementOpen);
1362  }
1363  case State::StatementStartNoLstrip: {
1364  state = State::StatementBody;
1365  pos += config.statement_open_no_lstrip.size();
1366  return make_token(Token::Kind::StatementOpen);
1367  }
1368  case State::StatementStartForceLstrip: {
1369  state = State::StatementBody;
1370  pos += config.statement_open_force_lstrip.size();
1371  return make_token(Token::Kind::StatementOpen);
1372  }
1373  case State::CommentStart: {
1374  state = State::CommentBody;
1375  pos += config.comment_open.size();
1376  return make_token(Token::Kind::CommentOpen);
1377  }
1378  case State::CommentStartForceLstrip: {
1379  state = State::CommentBody;
1380  pos += config.comment_open_force_lstrip.size();
1381  return make_token(Token::Kind::CommentOpen);
1382  }
1383  case State::ExpressionBody:
1384  return scan_body(config.expression_close, Token::Kind::ExpressionClose, config.expression_close_force_rstrip);
1385  case State::LineBody:
1386  return scan_body("\n", Token::Kind::LineStatementClose);
1387  case State::StatementBody:
1388  return scan_body(config.statement_close, Token::Kind::StatementClose, config.statement_close_force_rstrip, config.trim_blocks);
1389  case State::CommentBody: {
1390  // fast-scan to comment close
1391  const size_t end = m_in.substr(pos).find(config.comment_close);
1392  if (end == std::string_view::npos) {
1393  pos = m_in.size();
1394  return make_token(Token::Kind::Eof);
1395  }
1396 
1397  // Check for trim pattern
1398  const bool must_rstrip = inja::string_view::starts_with(m_in.substr(pos + end - 1), config.comment_close_force_rstrip);
1399 
1400  // return the entire comment in the close token
1401  state = State::Text;
1402  pos += end + config.comment_close.size();
1403  Token tok = make_token(Token::Kind::CommentClose);
1404 
1405  if (must_rstrip || config.trim_blocks) {
1406  skip_whitespaces_and_first_newline();
1407  }
1408  return tok;
1409  }

1555  std::shared_ptr<ExpressionNode> parse_expression(Template& tmpl) {
1556  size_t current_bracket_level {0};
1557  size_t current_brace_level {0};
1558  Arguments arguments;
1559  OperatorStack operator_stack;
1560 
1561  while (tok.kind != Token::Kind::Eof) {
1562  // Literals
1563  switch (tok.kind) {
1564  case Token::Kind::String: {
1565  if (current_brace_level == 0 && current_bracket_level == 0) {
1566  literal_start = tok.text;
1567  add_literal(arguments, tmpl.content.c_str());
1568  }
1569  } break;
1570  case Token::Kind::Number: {
1571  if (current_brace_level == 0 && current_bracket_level == 0) {
1572  literal_start = tok.text;
1573  add_literal(arguments, tmpl.content.c_str());
1574  }
1575  } break;
1576  case Token::Kind::LeftBracket: {
1577  if (current_brace_level == 0 && current_bracket_level == 0) {
1578  literal_start = tok.text;
1579  }
1580  current_bracket_level += 1;
1581  } break;
1582  case Token::Kind::LeftBrace: {
1583  if (current_brace_level == 0 && current_bracket_level == 0) {
1584  literal_start = tok.text;
1585  }
1586  current_brace_level += 1;
1587  } break;
1588  case Token::Kind::RightBracket: {
1589  if (current_bracket_level == 0) {
1590  throw_parser_error("unexpected ']'");
1591  }
1592 
1593  current_bracket_level -= 1;
1594  if (current_brace_level == 0 && current_bracket_level == 0) {
1595  add_literal(arguments, tmpl.content.c_str());
1596  }
1597  } break;
1598  case Token::Kind::RightBrace: {
1599  if (current_brace_level == 0) {
1600  throw_parser_error("unexpected '}'");
1601  }
1602 
1603  current_brace_level -= 1;
1604  if (current_brace_level == 0 && current_bracket_level == 0) {
1605  add_literal(arguments, tmpl.content.c_str());
1606  }
1607  } break;
1608  case Token::Kind::Id: {
1609  get_peek_token();
1610 
1611  // Data Literal
1612  if (tok.text == static_cast<decltype(tok.text)>("true") || tok.text == static_cast<decltype(tok.text)>("false") ||
1613  tok.text == static_cast<decltype(tok.text)>("null")) {
1614  if (current_brace_level == 0 && current_bracket_level == 0) {
1615  literal_start = tok.text;
1616  add_literal(arguments, tmpl.content.c_str());
1617  }
1618 
1619  // Operator
1620  } else if (tok.text == "and" || tok.text == "or" || tok.text == "in" || tok.text == "not") {
1621  goto parse_operator;
1622 
1623  // Functions
1624  } else if (peek_tok.kind == Token::Kind::LeftParen) {
1625  auto func = std::make_shared<FunctionNode>(tok.text, tok.text.data() - tmpl.content.c_str());
1626  get_next_token();
1627  do {
1628  get_next_token();
1629  auto expr = parse_expression(tmpl);
1630  if (!expr) {
1631  break;
1632  }
1633  func->number_args += 1;
1634  func->arguments.emplace_back(expr);
1635  } while (tok.kind == Token::Kind::Comma);
1636  if (tok.kind != Token::Kind::RightParen) {
1637  throw_parser_error("expected right parenthesis, got '" + tok.describe() + "'");
1638  }
1639 
1640  auto function_data = function_storage.find_function(func->name, func->number_args);
1641  if (function_data.operation == FunctionStorage::Operation::None) {
1642  throw_parser_error("unknown function " + func->name);
1643  }
1644  func->operation = function_data.operation;
1645  if (function_data.operation == FunctionStorage::Operation::Callback) {
1646  func->callback = function_data.callback;
1647  }
1648  arguments.emplace_back(func);
1649 
1650  // Variables
1651  } else {
1652  arguments.emplace_back(std::make_shared<DataNode>(static_cast<std::string>(tok.text), tok.text.data() - tmpl.content.c_str()));
1653  }
1654 
1655  // Operators
1656  } break;
1657  case Token::Kind::Equal:
1658  case Token::Kind::NotEqual:
1659  case Token::Kind::GreaterThan:
1660  case Token::Kind::GreaterEqual:
1661  case Token::Kind::LessThan:
1662  case Token::Kind::LessEqual:
1663  case Token::Kind::Plus:
1664  case Token::Kind::Minus:
1665  case Token::Kind::Times:
1666  case Token::Kind::Slash:
1667  case Token::Kind::Power:
1668  case Token::Kind::Percent:
1669  case Token::Kind::Dot: {
1670 
1671  parse_operator:
1672  FunctionStorage::Operation operation;
1673  switch (tok.kind) {
1674  case Token::Kind::Id: {
1675  if (tok.text == "and") {
1676  operation = FunctionStorage::Operation::And;
1677  } else if (tok.text == "or") {
1678  operation = FunctionStorage::Operation::Or;
1679  } else if (tok.text == "in") {
1680  operation = FunctionStorage::Operation::In;
1681  } else if (tok.text == "not") {
1682  operation = FunctionStorage::Operation::Not;
1683  } else {
1684  throw_parser_error("unknown operator in parser.");
1685  }
1686  } break;
1687  case Token::Kind::Equal: {
1688  operation = FunctionStorage::Operation::Equal;
1689  } break;
1690  case Token::Kind::NotEqual: {
1691  operation = FunctionStorage::Operation::NotEqual;
1692  } break;
1693  case Token::Kind::GreaterThan: {
1694  operation = FunctionStorage::Operation::Greater;
1695  } break;
1696  case Token::Kind::GreaterEqual: {
1697  operation = FunctionStorage::Operation::GreaterEqual;
1698  } break;
1699  case Token::Kind::LessThan: {
1700  operation = FunctionStorage::Operation::Less;
1701  } break;
1702  case Token::Kind::LessEqual: {
1703  operation = FunctionStorage::Operation::LessEqual;
1704  } break;
1705  case Token::Kind::Plus: {
1706  operation = FunctionStorage::Operation::Add;
1707  } break;
1708  case Token::Kind::Minus: {
1709  operation = FunctionStorage::Operation::Subtract;
1710  } break;
1711  case Token::Kind::Times: {
1712  operation = FunctionStorage::Operation::Multiplication;
1713  } break;
1714  case Token::Kind::Slash: {
1715  operation = FunctionStorage::Operation::Division;
1716  } break;
1717  case Token::Kind::Power: {
1718  operation = FunctionStorage::Operation::Power;
1719  } break;
1720  case Token::Kind::Percent: {
1721  operation = FunctionStorage::Operation::Modulo;
1722  } break;
1723  case Token::Kind::Dot: {
1724  operation = FunctionStorage::Operation::AtId;
1725  } break;
1726  default: {
1727  throw_parser_error("unknown operator in parser.");
1728  }
1729  }
1730  auto function_node = std::make_shared<FunctionNode>(operation, tok.text.data() - tmpl.content.c_str());
1731 
1732  while (!operator_stack.empty() &&
1733  ((operator_stack.top()->precedence > function_node->precedence) ||
1734  (operator_stack.top()->precedence == function_node->precedence && function_node->associativity == FunctionNode::Associativity::Left))) {
1735  add_operator(arguments, operator_stack);
1736  }
1737 
1738  operator_stack.emplace(function_node);
1739  } break;
1740  case Token::Kind::Comma: {
1741  if (current_brace_level == 0 && current_bracket_level == 0) {
1742  goto break_loop;
1743  }
1744  } break;
1745  case Token::Kind::Colon: {
1746  if (current_brace_level == 0 && current_bracket_level == 0) {
1747  throw_parser_error("unexpected ':'");
1748  }
1749  } break;
1750  case Token::Kind::LeftParen: {
1751  get_next_token();
1752  auto expr = parse_expression(tmpl);
1753  if (tok.kind != Token::Kind::RightParen) {
1754  throw_parser_error("expected right parenthesis, got '" + tok.describe() + "'");
1755  }
1756  if (!expr) {
1757  throw_parser_error("empty expression in parentheses");
1758  }
1759  arguments.emplace_back(expr);
1760  } break;
1761  default:
1762  goto break_loop;
1763  }
1764 
1765  get_next_token();
1766  }
1767 
1768  break_loop:
1769  while (!operator_stack.empty()) {
1770  add_operator(arguments, operator_stack);
1771  }
1772 
1773  std::shared_ptr<ExpressionNode> expr;
1774  if (arguments.size() == 1) {
1775  expr = arguments[0];
1776  arguments = {};
1777  } else if (arguments.size() > 1) {
1778  throw_parser_error("malformed expression");
1779  }
1780  return expr;
1781  }

1783  bool parse_statement(Template& tmpl, Token::Kind closing, std::string_view path) {
1784  if (tok.kind != Token::Kind::Id) {
1785  return false;
1786  }
1787 
1788  if (tok.text == static_cast<decltype(tok.text)>("if")) {
1789  get_next_token();
1790 
1791  auto if_statement_node = std::make_shared<IfStatementNode>(current_block, tok.text.data() - tmpl.content.c_str());
1792  current_block->nodes.emplace_back(if_statement_node);
1793  if_statement_stack.emplace(if_statement_node.get());
1794  current_block = &if_statement_node->true_statement;
1795  current_expression_list = &if_statement_node->condition;
1796 
1797  if (!parse_expression(tmpl, closing)) {
1798  return false;
1799  }
1800  } else if (tok.text == static_cast<decltype(tok.text)>("else")) {
1801  if (if_statement_stack.empty()) {
1802  throw_parser_error("else without matching if");
1803  }
1804  auto& if_statement_data = if_statement_stack.top();
1805  get_next_token();
1806 
1807  if_statement_data->has_false_statement = true;
1808  current_block = &if_statement_data->false_statement;
1809 
1810  // Chained else if
1811  if (tok.kind == Token::Kind::Id && tok.text == static_cast<decltype(tok.text)>("if")) {
1812  get_next_token();
1813 
1814  auto if_statement_node = std::make_shared<IfStatementNode>(true, current_block, tok.text.data() - tmpl.content.c_str());
1815  current_block->nodes.emplace_back(if_statement_node);
1816  if_statement_stack.emplace(if_statement_node.get());
1817  current_block = &if_statement_node->true_statement;
1818  current_expression_list = &if_statement_node->condition;
1819 
1820  if (!parse_expression(tmpl, closing)) {
1821  return false;
1822  }
1823  }
1824  } else if (tok.text == static_cast<decltype(tok.text)>("endif")) {
1825  if (if_statement_stack.empty()) {
1826  throw_parser_error("endif without matching if");
1827  }
1828 
1829  // Nested if statements
1830  while (if_statement_stack.top()->is_nested) {
1831  if_statement_stack.pop();
1832  }
1833 
1834  auto& if_statement_data = if_statement_stack.top();
1835  get_next_token();
1836 
1837  current_block = if_statement_data->parent;
1838  if_statement_stack.pop();
1839  } else if (tok.text == static_cast<decltype(tok.text)>("block")) {
1840  get_next_token();
1841 
1842  if (tok.kind != Token::Kind::Id) {
1843  throw_parser_error("expected block name, got '" + tok.describe() + "'");
1844  }
1845 
1846  const std::string block_name = static_cast<std::string>(tok.text);
1847 
1848  auto block_statement_node = std::make_shared<BlockStatementNode>(current_block, block_name, tok.text.data() - tmpl.content.c_str());
1849  current_block->nodes.emplace_back(block_statement_node);
1850  block_statement_stack.emplace(block_statement_node.get());
1851  current_block = &block_statement_node->block;
1852  auto success = tmpl.block_storage.emplace(block_name, block_statement_node);
1853  if (!success.second) {
1854  throw_parser_error("block with the name '" + block_name + "' does already exist");
1855  }
1856 
1857  get_next_token();
1858  } else if (tok.text == static_cast<decltype(tok.text)>("endblock")) {
1859  if (block_statement_stack.empty()) {
1860  throw_parser_error("endblock without matching block");
1861  }
1862 
1863  auto& block_statement_data = block_statement_stack.top();
1864  get_next_token();
1865 
1866  current_block = block_statement_data->parent;
1867  block_statement_stack.pop();
1868  } else if (tok.text == static_cast<decltype(tok.text)>("for")) {
1869  get_next_token();
1870 
1871  // options: for a in arr; for a, b in obj
1872  if (tok.kind != Token::Kind::Id) {
1873  throw_parser_error("expected id, got '" + tok.describe() + "'");
1874  }
1875 
1876  Token value_token = tok;
1877  get_next_token();
1878 
1879  // Object type
1880  std::shared_ptr<ForStatementNode> for_statement_node;
1881  if (tok.kind == Token::Kind::Comma) {
1882  get_next_token();
1883  if (tok.kind != Token::Kind::Id) {
1884  throw_parser_error("expected id, got '" + tok.describe() + "'");
1885  }
1886 
1887  Token key_token = std::move(value_token);
1888  value_token = tok;
1889  get_next_token();
1890 
1891  for_statement_node = std::make_shared<ForObjectStatementNode>(static_cast<std::string>(key_token.text), static_cast<std::string>(value_token.text),
1892  current_block, tok.text.data() - tmpl.content.c_str());
1893 
1894  // Array type
1895  } else {
1896  for_statement_node =
1897  std::make_shared<ForArrayStatementNode>(static_cast<std::string>(value_token.text), current_block, tok.text.data() - tmpl.content.c_str());
1898  }
1899 
1900  current_block->nodes.emplace_back(for_statement_node);
1901  for_statement_stack.emplace(for_statement_node.get());
1902  current_block = &for_statement_node->body;
1903  current_expression_list = &for_statement_node->condition;
1904 
1905  if (tok.kind != Token::Kind::Id || tok.text != static_cast<decltype(tok.text)>("in")) {
1906  throw_parser_error("expected 'in', got '" + tok.describe() + "'");
1907  }
1908  get_next_token();
1909 
1910  if (!parse_expression(tmpl, closing)) {
1911  return false;
1912  }
1913  } else if (tok.text == static_cast<decltype(tok.text)>("endfor")) {
1914  if (for_statement_stack.empty()) {
1915  throw_parser_error("endfor without matching for");
1916  }
1917 
1918  auto& for_statement_data = for_statement_stack.top();
1919  get_next_token();
1920 
1921  current_block = for_statement_data->parent;
1922  for_statement_stack.pop();
1923  } else if (tok.text == static_cast<decltype(tok.text)>("include")) {
1924  get_next_token();
1925 
1926  std::string template_name = parse_filename();
1927  add_to_template_storage(path, template_name);
1928 
1929  current_block->nodes.emplace_back(std::make_shared<IncludeStatementNode>(template_name, tok.text.data() - tmpl.content.c_str()));
1930 
1931  get_next_token();
1932  } else if (tok.text == static_cast<decltype(tok.text)>("extends")) {
1933  get_next_token();
1934 
1935  std::string template_name = parse_filename();
1936  add_to_template_storage(path, template_name);
1937 
1938  current_block->nodes.emplace_back(std::make_shared<ExtendsStatementNode>(template_name, tok.text.data() - tmpl.content.c_str()));
1939 
1940  get_next_token();
1941  } else if (tok.text == static_cast<decltype(tok.text)>("set")) {
1942  get_next_token();
1943 
1944  if (tok.kind != Token::Kind::Id) {
1945  throw_parser_error("expected variable name, got '" + tok.describe() + "'");
1946  }
1947 
1948  std::string key = static_cast<std::string>(tok.text);
1949  get_next_token();
1950 
1951  auto set_statement_node = std::make_shared<SetStatementNode>(key, tok.text.data() - tmpl.content.c_str());
1952  current_block->nodes.emplace_back(set_statement_node);
1953  current_expression_list = &set_statement_node->expression;
1954 
1955  if (tok.text != static_cast<decltype(tok.text)>("=")) {
1956  throw_parser_error("expected '=', got '" + tok.describe() + "'");
1957  }
1958  get_next_token();
1959 
1960  if (!parse_expression(tmpl, closing)) {
1961  return false;
1962  }
1963  } else {
1964  return false;
1965  }
1966  return true;
1967  }

1969  void parse_into(Template& tmpl, std::string_view path) {
1970  lexer.start(tmpl.content);
1971  current_block = &tmpl.root;
1972 
1973  for (;;) {
1974  get_next_token();
1975  switch (tok.kind) {
1976  case Token::Kind::Eof: {
1977  if (!if_statement_stack.empty()) {
1978  throw_parser_error("unmatched if");
1979  }
1980  if (!for_statement_stack.empty()) {
1981  throw_parser_error("unmatched for");
1982  }
1983  }
1984  return;
1985  case Token::Kind::Text: {
1986  current_block->nodes.emplace_back(std::make_shared<TextNode>(tok.text.data() - tmpl.content.c_str(), tok.text.size()));
1987  } break;
1988  case Token::Kind::StatementOpen: {
1989  get_next_token();
1990  if (!parse_statement(tmpl, Token::Kind::StatementClose, path)) {
1991  throw_parser_error("expected statement, got '" + tok.describe() + "'");
1992  }
1993  if (tok.kind != Token::Kind::StatementClose) {
1994  throw_parser_error("expected statement close, got '" + tok.describe() + "'");
1995  }
1996  } break;
1997  case Token::Kind::LineStatementOpen: {
1998  get_next_token();
1999  if (!parse_statement(tmpl, Token::Kind::LineStatementClose, path)) {
2000  throw_parser_error("expected statement, got '" + tok.describe() + "'");
2001  }
2002  if (tok.kind != Token::Kind::LineStatementClose && tok.kind != Token::Kind::Eof) {
2003  throw_parser_error("expected line statement close, got '" + tok.describe() + "'");
2004  }
2005  } break;
2006  case Token::Kind::ExpressionOpen: {
2007  get_next_token();
2008 
2009  auto expression_list_node = std::make_shared<ExpressionListNode>(tok.text.data() - tmpl.content.c_str());
2010  current_block->nodes.emplace_back(expression_list_node);
2011  current_expression_list = expression_list_node.get();
2012 
2013  if (!parse_expression(tmpl, Token::Kind::ExpressionClose)) {
2014  throw_parser_error("expected expression close, got '" + tok.describe() + "'");
2015  }
2016  } break;
2017  case Token::Kind::CommentOpen: {
2018  get_next_token();
2019  if (tok.kind != Token::Kind::CommentClose) {
2020  throw_parser_error("expected comment close, got '" + tok.describe() + "'");
2021  }
2022  } break;
2023  default: {
2024  throw_parser_error("unexpected token '" + tok.describe() + "'");
2025  } break;
2026  }
2027  }
