53  void load(const sv path)
54  {
55  const std::filesystem::path libpath(path);
56  if (std::filesystem::is_directory(libpath)) {
57  load_libs(path);
58  }
59  else if (libpath.extension() == SHARED_LIBRARY_EXTENSION) {
60  load_lib(libpath.string());
61  }
62  else {
63  load_lib_by_name(libpath.string());
64  }
65  }

67  void load_libs(const sv directory)
68  {
69  std::filesystem::directory_entry dir(directory);
70  for (const auto& entry : std::filesystem::directory_iterator(dir)) {
71  if (entry.is_regular_file() && entry.path().extension() == SHARED_LIBRARY_EXTENSION) {
72  load_lib(entry.path().string());
73  }
74  }
75  }

101  {

104 #else
105  lib_t loaded_lib = dlopen(path.c_str(), RTLD_LAZY);
106 #endif
107  if (loaded_lib) {
108  loaded_libs.emplace_back(loaded_lib);
109 
110 #ifdef GLAZE_API_ON_WINDOWS
111 #ifdef __GNUC__
112 #pragma GCC diagnostic push
113 #pragma GCC diagnostic ignored "-Wcast-function-type"
114  auto* ptr = reinterpret_cast<create>(GetProcAddress(loaded_lib, "glz_iface"));
115 #pragma GCC diagnostic pop
116 #else
117  auto* ptr = reinterpret_cast<create>(GetProcAddress(loaded_lib, "glz_iface"));
118 #endif
119 #else
120  auto* ptr = reinterpret_cast<create>(dlsym(dlopen(path.c_str(), RTLD_NOW), "glz_iface"));
121 #endif
122  if (ptr) {
123  std::shared_ptr<glz::iface> shared_iface_ptr = (*ptr)()();
124  api_map.merge(*shared_iface_ptr);
125  return true;
126  }
127  }
