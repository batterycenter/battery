20  GLZ_ALWAYS_INLINE void u128_mul(uint64_t a, uint64_t b, uint64_t* hi, uint64_t* lo) noexcept
21  {
22 #ifdef __SIZEOF_INT128__
23 #if defined(__GNUC__) || defined(__GNUG__)
24 #pragma GCC diagnostic push
25 #pragma GCC diagnostic ignored "-Wpedantic"
26 #endif
27  unsigned __int128 m = static_cast<unsigned __int128>(a) * b;
28 #if defined(__GNUC__) || defined(__GNUG__)
29 #pragma GCC diagnostic pop
30 #endif
31  *hi = uint64_t(m >> 64);
32  *lo = uint64_t(m);
33 #elif defined(_M_X64)
34  *lo = _umul128(a, b, hi);
35 #elif defined(_M_ARM64)
36  *hi = __umulh(a, b);
37  *lo = a * b;
38 #else
39  uint32_t a0 = (uint32_t)(a), a1 = (uint32_t)(a >> 32);
40  uint32_t b0 = (uint32_t)(b), b1 = (uint32_t)(b >> 32);
41  uint64_t p00 = (uint64_t)a0 * b0, p01 = (uint64_t)a0 * b1;
42  uint64_t p10 = (uint64_t)a1 * b0, p11 = (uint64_t)a1 * b1;
43  uint64_t m0 = p01 + (p00 >> 32);
44  uint32_t m00 = (uint32_t)(m0), m01 = (uint32_t)(m0 >> 32);
45  uint64_t m1 = p10 + m00;
46  uint32_t m10 = (uint32_t)(m1), m11 = (uint32_t)(m1 >> 32);
47  *hi = p11 + m01 + m11;
48  *lo = ((uint64_t)m10 << 32) | (uint32_t)p00;
49 #endif
50  }

62 #if defined(__GNUC__) || defined(__GNUG__)

67 #else

69  u128_mul(a, b, &h, &l);

73  *lo = t;
74 #endif
