53  inline uint64_t mulhi64(uint64_t a, uint64_t b)
54  {
55 #if defined(__GNUC__) || defined(__GNUG__)
56 #pragma GCC diagnostic push
57 #pragma GCC diagnostic ignored "-Wpedantic"
58  unsigned __int128 prod = a * static_cast<unsigned __int128>(b);
59 #pragma GCC diagnostic pop
60 #else
61  unsigned __int128 prod = a * static_cast<unsigned __int128>(b);
62 #endif
63  return prod >> 64;
64  }
65 #elif defined(_M_X64) || defined(_M_ARM64)
66 #define mulhi64 __umulh
67 #else
68  uint64_t mulhi64(uint64_t a, uint64_t b) noexcept
69  {
70  uint64_t a_lo = (uint32_t)a;
71  uint64_t a_hi = a >> 32;
72  uint64_t b_lo = (uint32_t)b;
73  uint64_t b_hi = b >> 32;
74  uint64_t a_x_b_hi = a_hi * b_hi;
75  uint64_t a_x_b_mid = a_hi * b_lo;
76  uint64_t b_x_a_mid = b_hi * a_lo;
77  uint64_t a_x_b_lo = a_lo * b_lo;
78  uint64_t carry_bit = ((uint64_t)(uint32_t)a_x_b_mid + (uint64_t)(uint32_t)b_x_a_mid + (a_x_b_lo >> 32)) >> 32;
79  uint64_t multhi = a_x_b_hi + (a_x_b_mid >> 32) + (b_x_a_mid >> 32) + carry_bit;
80  return multhi;
81  }

335  void mul_u32(uint32_t num) noexcept
336  {
337  uint32_t carry = 0;
338  for (std::size_t i = 0; i < data.size(); i++) {
339  uint64_t res = uint64_t(data[i]) * uint64_t(num) + uint64_t(carry);
340  uint32_t lower_word = uint32_t(res);
341  uint32_t upper_word = uint32_t(res >> 32);
342  data[i] = lower_word;
343  carry = upper_word;
344  }
345  if (carry != 0) {
346  data.emplace_back(carry);
347  }

360  void mul_pow2(uint32_t exp) noexcept
361  {
362  uint32_t shft = exp % 32;
363  uint32_t move = exp / 32;
364  uint32_t idx = static_cast<uint32_t>(data.size()) - 1;
365  if (shft == 0) {
366  data.resize(data.size() + move);
367  for (; idx > 0; idx--) {
368  data[idx + move - 1] = data[idx - 1];
369  }
370  while (move) data[--move] = 0;
371  }
372  else {
373  data.resize(data.size() + move + 1);
374  ++idx;
375  for (; idx > 0; idx--) {
376  uint32_t num = data[idx] << shft;
377  num |= data[idx - 1] >> (32 - shft);
378  data[idx + move] = num;
379  }
380  data[move] = data[0] << shft;
381  if (data.back() == 0) data.pop_back();
382  while (move) data[--move] = 0;
383  }

399  template <std::floating_point T, bool force_conformance = false>
400  inline bool parse_float(T& val, auto*& cur) noexcept
401  {
402  const uint8_t* sig_cut = nullptr; /* significant part cutting position for long number */
403  [[maybe_unused]] const uint8_t* sig_end = nullptr; /* significant part ending position */
404  const uint8_t* dot_pos = nullptr; /* decimal point position */
405  uint32_t frac_zeros = 0;
406  uint64_t sig = 0; /* significant part of the number */
407  int32_t exp = 0; /* exponent part of the number */
408  bool exp_sign; /* temporary exponent sign from literal part */
409  int32_t exp_sig = 0; /* temporary exponent number from significant part */
410  int32_t exp_lit = 0; /* temporary exponent number from exponent literal part */
411  uint64_t num_tmp; /* temporary number for reading */
412  const uint8_t* tmp; /* temporary cursor for reading */
413  const uint8_t* hdr = cur;
414  bool sign = (*hdr == '-');
415  cur += sign;
416  auto apply_sign = [&](auto&& val) -> T { return sign ? -static_cast<T>(val) : static_cast<T>(val); };
417  /* begin with non-zero digit */
418  sig = uint64_t(*cur - '0');
419  if (sig > 9) {
420  if constexpr (std::integral<T>) {
421  return false;
422  }
423  else if (*cur == 'n' && cur[1] == 'u' && cur[2] == 'l' && cur[3] == 'l') {
424  cur += 4;
425  val = std::numeric_limits<T>::quiet_NaN();
426  return true;
427  }
428  else if ((*cur | e_bit) == 'n' && (cur[1] | e_bit) == 'a' && (cur[2] | e_bit) == 'n') {
429  cur += 3;
430  val = sign ? -std::numeric_limits<T>::quiet_NaN() : std::numeric_limits<T>::quiet_NaN();
431  return true;
432  }
433  else {
434  return false;
435  }
436  }
437  constexpr auto zero = static_cast<uint8_t>('0');
438 #define expr_intg(i) \
439  if ((num_tmp = cur[i] - zero) <= 9) [[likely]] \
440  sig = num_tmp + sig * 10; \
441  else { \
442  if constexpr (force_conformance && i > 1) { \
443  if (*cur == zero) return false; \
444  } \
445  goto digi_sepr_##i; \
446  }
447  repeat_in_1_18(expr_intg);
448 #undef expr_intg
449  if (*cur == zero) [[unlikely]] {
450  return false;
451  }
452  cur += 19; /* skip continuous 19 digits */
453  if (!digi_is_digit_or_fp(*cur)) {
454  val = static_cast<T>(sig);
455  if constexpr (!std::is_unsigned_v<T>) {
456  val *= sign ? -1 : 1;
457  }
458  return true;
459  }
460  goto digi_intg_more; /* read more digits in integral part */
461  /* process first non-digit character */
462 #define expr_sepr(i) \
463  digi_sepr_##i : if ((!digi_is_fp(cur[i]))) [[likely]] \
464  { \
465  cur += i; \
466  val = apply_sign(sig); \
467  return true; \
468  } \
469  dot_pos = cur + i; \
470  if ((cur[i] == '.')) [[likely]] { \
471  if (sig == 0) \
472  while (cur[frac_zeros + i + 1] == zero) ++frac_zeros; \
473  goto digi_frac_##i; \
474  } \
475  cur += i; \
476  sig_end = cur; \
477  goto digi_exp_more;
478  repeat_in_1_18(expr_sepr)
479 #undef expr_sepr
480  /* read fraction part */
481 #define expr_frac(i) \
482  digi_frac_##i : if (((num_tmp = static_cast<uint64_t>(cur[i + 1 + frac_zeros] - zero)) <= 9)) [[likely]] sig = \
483  num_tmp + sig * 10; \
484  else \
485  { \
486  goto digi_stop_##i; \
487  }
488  repeat_in_1_18(expr_frac)
489 #undef expr_frac
490  cur += 20 + frac_zeros; /* skip 19 digits and 1 decimal point */
491  if (uint8_t(*cur - zero) > 9) goto digi_frac_end; /* fraction part end */
492  goto digi_frac_more; /* read more digits in fraction part */
493  /* significant part end */
494 #define expr_stop(i) \
495  digi_stop_##i : cur += i + 1 + frac_zeros; \
496  goto digi_frac_end;
497  repeat_in_1_18(expr_stop)
498 #undef expr_stop
499  /* read more digits in integral part */
500  digi_intg_more : static constexpr uint64_t U64_MAX = (std::numeric_limits<uint64_t>::max)(); // todo
501  if ((num_tmp = *cur - zero) < 10) {
502  if (!digi_is_digit_or_fp(cur[1])) {
503  /* this number is an integer consisting of 20 digits */
504  if ((sig < (U64_MAX / 10)) || (sig == (U64_MAX / 10) && num_tmp <= (U64_MAX % 10))) {
505  sig = num_tmp + sig * 10;
506  cur++;
507  val = static_cast<T>(sig);
508  if constexpr (!std::is_unsigned_v<T>) {
509  val *= sign ? -1 : 1;
510  }
511  return true;
512  }
513  }
514  }
515  if ((e_bit | *cur) == 'e') {
516  dot_pos = cur;
517  goto digi_exp_more;
518  }
519  if (*cur == '.') {
520  dot_pos = cur++;
521  if (uint8_t(*cur - zero) > 9) {
522  return false;
523  }
524  }
525  /* read more digits in fraction part */
526  digi_frac_more:
527  sig_cut = cur; /* too large to fit in u64, excess digits need to be cut */
528  sig += (*cur >= '5'); /* round */
529  while (uint8_t(*++cur - zero) < 10) {
530  }
531  if (!dot_pos) {
532  dot_pos = cur;
533  if (*cur == '.') {
534  if (uint8_t(*++cur - zero) > 9) {
535  return false;
536  }
537  while (uint8_t(*++cur - zero) < 10) {
538  }
539  }
540  }
541  exp_sig = static_cast<int32_t>(dot_pos - sig_cut);
542  exp_sig += (dot_pos < sig_cut);
543  // ignore trailing zeros
544  tmp = cur - 1;
545  while (*tmp == '0' || *tmp == '.') {
546  --tmp;
547  }
548  if (tmp < sig_cut) {
549  sig_cut = nullptr;
550  }
551  else {
552  sig_end = cur;
553  }
554  if ((e_bit | *cur) == 'e') goto digi_exp_more;
555  goto digi_exp_finish;
556  // fraction part end
557  digi_frac_end:
558  sig_end = cur;
559  exp_sig = -int32_t((cur - dot_pos) - 1);
560  if constexpr (force_conformance) {
561  if (exp_sig == 0) return false;
562  }
563  if ((e_bit | *cur) != 'e') [[likely]] {
564  if ((exp_sig < F64_MIN_DEC_EXP - 19)) [[unlikely]] {
565  val = apply_sign(0);
566  return true;
567  }
568  exp = exp_sig;
569  goto digi_finish;
570  }
571  else {
572  goto digi_exp_more;
573  }
574  // read exponent part
575  digi_exp_more:
576  exp_sign = (*++cur == '-');
577  cur += (*cur == '+' || *cur == '-');
578  if (uint8_t(*cur - zero) > 9) [[unlikely]] {
579  if constexpr (force_conformance) {
580  return false;
581  }
582  else {
583  goto digi_finish;
584  }
585  }
586  while (*cur == '0') {
587  ++cur;
588  }
589  // read exponent literal
590  tmp = cur;
591  uint8_t c;
592  while (uint8_t(c = *cur - zero) < 10) {
593  ++cur;
594  exp_lit = c + uint32_t(exp_lit) * 10;
595  }
596  // large exponent case
597  if ((cur - tmp >= 6)) [[unlikely]] {
598  if (sig == 0 || exp_sign) {
599  val = apply_sign(0);
600  val = static_cast<T>(sig);
601  return true;
602  }
603  else {
604  val = apply_sign(std::numeric_limits<T>::infinity());
605  return true;
606  }
607  }
608  exp_sig += exp_sign ? -exp_lit : exp_lit;
609  // validate exponent value
610  digi_exp_finish:
611  if (sig == 0) {
612  val = (sign ? -T{0} : T{0});
613  return true;
614  }
615  if ((exp_sig < F64_MIN_DEC_EXP - 19)) [[unlikely]] {
616  val = (sign ? -T{0} : T{0});
617  return true;
618  }
619  else if ((exp_sig > F64_MAX_DEC_EXP)) [[unlikely]] {
620  val = sign ? -std::numeric_limits<T>::infinity() : std::numeric_limits<T>::infinity();
621  return true;
622  }
623  exp = exp_sig;
624  // all digit read finished
625  digi_finish:
626 
627  if constexpr (std::is_same_v<double, T>) {
628  // numbers must be exactly representable in this fast path
629  if (sig < (uint64_t(1) << 53) && std::abs(exp) <= 22) {
630  val = static_cast<T>(sig);
631  if constexpr (!std::is_unsigned_v<T>) {
632  val *= sign ? -1 : 1;
633  }
634  if (exp >= 0) {
635  val *= powers_of_ten_float[exp];
636  }
637  else {
638  val /= powers_of_ten_float[-exp];
639  }
640  return true;
641  }
642  }
643  else {
644  if (sig < (uint64_t(1) << 24) && std::abs(exp) <= 8) {
645  val = static_cast<T>(sig);
646  if constexpr (!std::is_unsigned_v<T>) {
647  val *= sign ? -1 : 1;
