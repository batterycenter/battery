1127  template<typename TFunc>
1128  inline static std::expected<size_t, b::filesystem_error>
1129  try_read_chunked(const fs::path& path, size_t chunk_size, TFunc callback) {
1130  b::fs::ifstream file(path, std::ios::in | std::ios::binary);
1131  if (!file.is_open()) {
1132  return std::unexpected(b::filesystem_error(
1133  b::format("Failed loading file {}: {}", path, b::strerror(errno))
1134  ));
1135  }
1136 
1137  std::string buffer(chunk_size, 0);
1138  size_t totalBytes = 0;
1139  while (!file.eof()) {
1140  file.read(buffer.data(), buffer.size()); // Read a chunk of the file
1141  auto thisChunkSize = file.gcount();
1142  totalBytes += thisChunkSize;
1143 
1144  if (thisChunkSize != 0) {
1145  auto chunk = std::string(buffer.data(), static_cast<size_t>(thisChunkSize));
1146  callback(chunk);
1147  }
1148  }
1149 
1150  return totalBytes;
1151  }
