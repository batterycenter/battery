1168  template<typename encoding = b::enc::utf8, typename TFunc>
1169  inline static std::expected<size_t, b::filesystem_error> try_read_chunked(const fs::path& path,
1170  size_t chunk_size,
1171  TFunc callback) {
1172  std::ios::openmode filemode = std::ios::in;
1173  if constexpr (std::is_same_v<encoding, b::enc::binary>) {
1174  filemode |= std::ios::binary;
1175  }
1176 
1177  b::fs::ifstream file(path, filemode);
1178  if (!file.is_open()) {
1179  return std::unexpected(b::filesystem_error(
1180  b::format("Failed loading file {}: {}", path, b::strerror(errno))
1181  ));
1182  }
1183 
1184  std::string buffer(chunk_size, 0);
1185  size_t totalBytes = 0;
1186  while (!file.eof()) {
1187  file.read(buffer.data(), buffer.size()); // Read a chunk of the file
1188  auto thisChunkSize = file.gcount();
1189  totalBytes += thisChunkSize;
1190 
1191  if (thisChunkSize != 0) {
1192  auto chunk = b::string::decode<encoding>(
1193  std::string(buffer.data(), static_cast<size_t>(thisChunkSize))
1194  );
1195  callback(chunk);
1196  }
1197  }
1198 
1199  return std::expected(totalBytes);
