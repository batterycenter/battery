1133  template<typename TFunc>
1134  inline static std::expected<size_t, b::filesystem_error>
1135  try_read_chunked(const fs::path& path, size_t chunk_size, TFunc callback) {
1136  b::fs::ifstream file(path, std::ios::in | std::ios::binary);
1137  if (!file.is_open()) {
1138  return std::unexpected(b::filesystem_error(
1139  b::format("Failed loading file {}: {}", path.string(), b::strerror(errno))
1140  ));
1141  }
1142 
1143  std::string buffer(chunk_size, 0);
1144  size_t totalBytes = 0;
1145  while (!file.eof()) {
1146  file.read(buffer.data(), buffer.size()); // Read a chunk of the file
1147  auto thisChunkSize = file.gcount();
1148  totalBytes += thisChunkSize;
1149 
1150  if (thisChunkSize != 0) {
1151  auto chunk = std::string(buffer.data(), static_cast<size_t>(thisChunkSize));
1152  callback(chunk);
1153  }
1154  }
