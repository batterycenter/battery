190  inline bool isMsysPty(int fd) noexcept
191  {
192  // Dynamic load for binary compability with old Windows
193  const auto ptrGetFileInformationByHandleEx
194  = reinterpret_cast<decltype(&GetFileInformationByHandleEx)>(
195  GetProcAddress(GetModuleHandle(TEXT("kernel32.dll")),
196  "GetFileInformationByHandleEx"));
197  if (!ptrGetFileInformationByHandleEx) {
198  return false;
199  }
200 
201  HANDLE h = reinterpret_cast<HANDLE>(_get_osfhandle(fd));
202  if (h == INVALID_HANDLE_VALUE) {
203  return false;
204  }
205 
206  // Check that it's a pipe:
207  if (GetFileType(h) != FILE_TYPE_PIPE) {
208  return false;
209  }
210 
211  // POD type is binary compatible with FILE_NAME_INFO from WinBase.h
212  // It have the same alignment and used to avoid UB in caller code
213  struct MY_FILE_NAME_INFO {
214  DWORD FileNameLength;
215  WCHAR FileName[MAX_PATH];
216  };
217 
218  auto pNameInfo = std::unique_ptr<MY_FILE_NAME_INFO>(
219  new (std::nothrow) MY_FILE_NAME_INFO());
220  if (!pNameInfo) {
221  return false;
222  }
223 
224  // Check pipe name is template of
225  // {"cygwin-","msys-"}XXXXXXXXXXXXXXX-ptyX-XX
226  if (!ptrGetFileInformationByHandleEx(h, FileNameInfo, pNameInfo.get(),
227  sizeof(MY_FILE_NAME_INFO))) {
228  return false;
229  }
230  std::wstring name(pNameInfo->FileName, pNameInfo->FileNameLength / sizeof(WCHAR));
231  if ((name.find(L"msys-") == std::wstring::npos
232  && name.find(L"cygwin-") == std::wstring::npos)
233  || name.find(L"-pty") == std::wstring::npos) {
234  return false;
235  }
236 
237  return true;
238  }

345  inline const SGR &defaultState() noexcept
346  {
347  static const SGR defaultSgr = []() -> SGR {
348  CONSOLE_SCREEN_BUFFER_INFO info;
349  WORD attrib = FOREGROUND_RED | FOREGROUND_GREEN | FOREGROUND_BLUE;

424  inline WORD SGR2Attr(const SGR &state) noexcept
425  {
426  WORD attrib = 0;
427  if (state.conceal) {
428  if (state.inverse) {
429  attrib = (state.fgColor << 4) | state.fgColor;
430  if (state.bold)
431  attrib |= FOREGROUND_INTENSITY | BACKGROUND_INTENSITY;
432  } else {
433  attrib = (state.bgColor << 4) | state.bgColor;
434  if (state.underline)
435  attrib |= FOREGROUND_INTENSITY | BACKGROUND_INTENSITY;
436  }
437  } else if (state.inverse) {
438  attrib = (state.fgColor << 4) | state.bgColor;
439  if (state.bold) attrib |= BACKGROUND_INTENSITY;
440  if (state.underline) attrib |= FOREGROUND_INTENSITY;
441  } else {
442  attrib = state.fgColor | (state.bgColor << 4) | state.bold
443  | state.underline;
444  }
445  return attrib;
446  }
