46  class string_iterator {
47  public:

50  bool operator!=(const string_iterator& other) {

52  }

54  string_iterator operator++() {
55  if constexpr (Reverse) {
56  m_ptr--;
57  }
58  else {
59  m_ptr++;
60  }
61  return *this;
62  }
63 
64  string_iterator operator++(int) {
65  auto iter = *this;
66  if constexpr (Reverse) {
67  --(*this);
68  }
69  else {
70  ++(*this);
71  }
72  return iter;
73  }
74 
75  string_iterator operator--() {
76  if constexpr (Reverse) {
77  m_ptr++;
78  }
79  else {
80  m_ptr--;
81  }
82  return *this;
83  }
84 
85  string_iterator operator--(int) {
86  auto iter = *this;
87  if constexpr (Reverse) {
88  ++(*this);
89  }
90  else {
91  --(*this);
92  }
93  return iter;
94  }

106  }

110  if constexpr (Reverse) {
111  return m_ptr - 1;
