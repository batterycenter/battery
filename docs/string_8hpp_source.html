403  template<typename encoding, typename T>
404  static auto decode(const T& data) {
405  // !std::is_same_v<encoding, encoding> is used as an assertion trigger
406  // that only triggers when the specific if-branch is compiled
407 
408  // UTF-8
409  if constexpr (std::is_same_v<encoding, b::enc::utf8>) {
410  if constexpr (std::is_convertible_v<T, std::string>) { // std::string overload
411  return b::string::internal::decode_utf8(data);
412  }
413  else if constexpr (std::is_convertible_v<T, std::u8string>) { // std::u8string overload
414  return b::string::internal::decode_utf8(std::string(data.begin(), data.end()));
415  }
416  else if constexpr (std::is_convertible_v<T, b::bytearray>) { // b::bytearray overload
417  return b::string::internal::decode_utf8(data.str());
418  }
419  else {
420  static_assert(!std::is_same_v<encoding, encoding>,
421  "b::string::decode<>(): `b::enc::utf8` encoding may only take in "
422  "std::string, std::u8string and b::bytearray");
423  }
424  }
425 
426  // UTF-16
427  else if constexpr (std::is_same_v<encoding, b::enc::utf16>) {
428  if constexpr (std::is_convertible_v<T, std::u16string>) { // std::u16string overload
429  return b::string::internal::decode_utf16(data);
430  }
431  else if constexpr (std::is_convertible_v<T, std::wstring>) { // std::wstring overload
432  return b::string::internal::decode_utf16(std::u16string(data.begin(), data.end()));
433  }
434  else if constexpr (std::is_convertible_v<T, b::bytearray>) { // b::bytearray overload
435  return b::string::internal::decode_utf16(
436  std::u16string(std::bit_cast<const char16_t*>(data.data()), data.length() / 2)
437  );
438  }
439  else {
440  static_assert(!std::is_same_v<encoding, encoding>,
441  "b::string::decode<>(): `b::enc::utf16` encoding may only take in "
442  "std::u16string, std::wstring and b::bytearray");
443  }
444  }
445 
446  // UTF-16 widestring
447  else if constexpr (std::is_same_v<encoding, b::enc::utf16_wide>) {
448  if constexpr (std::is_convertible_v<T, std::u16string>) { // std::u16string overload
449  return b::string::internal::decode_utf16(data);
450  }
451  else if constexpr (std::is_convertible_v<T, std::wstring>) { // std::wstring overload
452  return b::string::internal::decode_utf16(std::u16string(data.begin(), data.end()));
453  }
454  else if constexpr (std::is_convertible_v<T, b::bytearray>) { // b::bytearray overload
455  return b::string::internal::decode_utf16(
456  std::u16string(std::bit_cast<const char16_t*>(data.data()), data.length() / 2)
457  );
458  }
459  else {
460  static_assert(!std::is_same_v<encoding, encoding>,
461  "b::string::decode<>(): `b::enc::utf16_wide` encoding may only take in "
462  "std::u16string, std::wstring and b::bytearray");
463  }
464  }
465 
466  // UTF-32
467  else if constexpr (std::is_same_v<encoding, b::enc::utf32>) {
468  if constexpr (std::is_convertible_v<T, std::u32string>) { // std::u32string overload
469  return b::string::internal::decode_utf32(data);
470  }
471  else if constexpr (std::is_same_v<T, char32_t>) { // char32_t overload
472  return b::string::internal::decode_utf32(std::u32string(&data, 1));

478  }

483  }
484  }

539  }
540  }

546 #else
547  return b::string::decode<b::enc::utf8>(data);

554  return b::bytearray::from_string(data);

561  }

670  }
