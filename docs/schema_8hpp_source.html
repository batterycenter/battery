308  template <glaze_object_t T>
309  struct to_json_schema<T>
310  {
311  template <auto Opts>
312  static void op(auto& s, auto& defs) noexcept
313  {
314  s.type = {"object"};
315 
316  using V = std::decay_t<T>;
317 
318  if constexpr (requires { meta<V>::required; }) {
319  s.required = meta<V>::required;
320  }
321 
322  if constexpr (requires { meta<V>::examples; }) {
323  s.examples = meta<V>::examples;
324  }
325 
326  static constexpr auto N = std::tuple_size_v<meta_t<V>>;
327  s.properties = std::map<std::string_view, schema, std::less<>>();
328  for_each<N>([&](auto I) {
329  static constexpr auto item = glz::tuplet::get<I>(meta_v<V>);
330  using mptr_t = decltype(glz::tuplet::get<1>(item));
331  using val_t = std::decay_t<member_t<V, mptr_t>>;
332  auto& def = defs[name_v<val_t>];
333  if (!def.type) {
334  to_json_schema<val_t>::template op<Opts>(def, defs);
335  }
336  auto ref_val = schema{join_v<chars<"#/$defs/">, name_v<val_t>>};
337  static constexpr auto Size = std::tuple_size_v<decltype(item)>;
338  if constexpr (Size > 2) {
339  using additional_data_type = decltype(glz::tuplet::get<2>(item));
340  if constexpr (std::is_convertible_v<additional_data_type, std::string_view>) {
341  ref_val.description = glz::tuplet::get<2>(item);
342  }
343  else if constexpr (std::is_convertible_v<additional_data_type, schema>) {
344  ref_val = glz::tuplet::get<2>(item);
345  ref_val.ref = join_v<chars<"#/$defs/">, name_v<val_t>>;
346  }
347  }
348  (*s.properties)[glz::tuplet::get<0>(item)] = ref_val;
349  });
350  s.additionalProperties = false;
351  }
352  };
