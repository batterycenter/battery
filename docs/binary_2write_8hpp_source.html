27  template <uint64_t i, class... Args>
28  GLZ_ALWAYS_INLINE void dump_compressed_int(Args&&... args) noexcept
29  {
30  if constexpr (i < 64) {
31  const uint8_t c = uint8_t(i) << 2;
32  dump_type(c, args...);
33  }
34  else if constexpr (i < 16384) {
35  const uint16_t c = uint16_t(1) | (uint16_t(i) << 2);
36  dump_type(c, args...);
37  }
38  else if constexpr (i < 1073741824) {
39  const uint32_t c = uint32_t(2) | (uint32_t(i) << 2);
40  dump_type(c, args...);
41  }
42  else if constexpr (i < 4611686018427387904) {
43  const uint64_t c = uint64_t(3) | (uint64_t(i) << 2);
44  dump_type(c, args...);
45  }
46  else {
47  static_assert(i >= 4611686018427387904, "size not supported");
48  }
49  }

276  template <writable_array_t T>
277  struct to_binary<T> final
278  {
279  template <auto Opts, class... Args>
280  GLZ_ALWAYS_INLINE static void op(auto&& value, is_context auto&& ctx, Args&&... args) noexcept
281  {
282  using V = range_value_t<std::decay_t<T>>;
283 
284  if constexpr (boolean_like<V>) {
285  constexpr uint8_t type = uint8_t(3) << 3;
286  constexpr uint8_t tag = tag::typed_array | type;
287  dump_type(tag, args...);
288  dump_compressed_int<Opts>(value.size(), args...);
289 
290  // booleans must be dumped using single bits
291  if constexpr (has_static_size<T>) {
292  constexpr auto num_bytes = (value.size() + 7) / 8;
293  std::array<uint8_t, num_bytes> bytes{};
294  for (size_t byte_i{}, i{}; byte_i < num_bytes - 1; ++byte_i) {
295  for (size_t bit_i = 7; bit_i < 8 && i < value.size(); --bit_i, ++i) {
296  bytes[byte_i] |= uint8_t(value[i]) << uint8_t(bit_i);
297  }
298  }
299  dump(bytes, args...);
300  }
301  else {
302  const auto num_bytes = (value.size() + 7) / 8;
303  for (size_t byte_i{}, i{}; byte_i < num_bytes; ++byte_i) {
304  uint8_t byte{};
305  for (size_t bit_i = 7; bit_i < 8 && i < value.size(); --bit_i, ++i) {
306  byte |= uint8_t(value[i]) << uint8_t(bit_i);
307  }
308  dump_type(byte, args...);
309  }
310  }
311  }
312  else if constexpr (num_t<V>) {
313  constexpr uint8_t type =
314  std::floating_point<V> ? 0 : (std::is_signed_v<V> ? 0b000'01'000 : 0b000'10'000);
315  constexpr uint8_t tag = tag::typed_array | type | (byte_count<V> << 5);
316  dump_type(tag, args...);
317  dump_compressed_int<Opts>(value.size(), args...);
318 
319  if constexpr (contiguous<T>) {
320  dump(std::as_bytes(std::span{value.data(), static_cast<size_t>(value.size())}), args...);
321  }
322  else {
323  for (auto& x : value) {
324  dump_type(x, args...);
325  }
326  }
327  }
328  else if constexpr (str_t<V>) {
329  constexpr uint8_t type = uint8_t(3) << 3;
330  constexpr uint8_t string_indicator = uint8_t(1) << 5;
331  constexpr uint8_t tag = tag::typed_array | type | string_indicator;
332  dump_type(tag, args...);
333  dump_compressed_int<Opts>(value.size(), args...);
334 
335  for (auto& x : value) {
336  dump_compressed_int<Opts>(x.size(), args...);
337  dump(std::as_bytes(std::span{x.data(), x.size()}), args...);
338  }
339  }
340  else if constexpr (complex_t<V>) {
341  constexpr uint8_t tag = tag::extensions | 0b00011'000;
342  dump_type(tag, args...);
343 
344  using X = typename V::value_type;
345  constexpr uint8_t complex_array = 1;
346  constexpr uint8_t type =
347  std::floating_point<X> ? 0 : (std::is_signed_v<X> ? 0b000'01'000 : 0b000'10'000);
348  constexpr uint8_t complex_header = complex_array | type | (byte_count<X> << 5);
349  dump_type(complex_header, args...);
350 
351  dump_compressed_int<Opts>(value.size(), args...);
352 
353  for (auto&& x : value) {
354  write<binary>::no_header<Opts>(x, ctx, args...);
355  }
356  }
357  else {
358  constexpr uint8_t tag = tag::generic_array;
359  dump_type(tag, args...);
360  dump_compressed_int<Opts>(value.size(), args...);
361 
362  for (auto&& x : value) {
363  write<binary>::op<Opts>(x, ctx, args...);
364  }
365  }
366  }
367  };

499  template <class Map, class Key>
500  concept findable = requires(Map& map, const Key& key) { map.find(key); };
501 
502  template <auto& Partial, opts Opts, class T, output_buffer Buffer>
503  [[nodiscard]] inline write_error write(T&& value, is_context auto&& ctx, Buffer& buffer, auto& ix) noexcept
504  {
505  write_error we{};
506 
507  if constexpr (std::count(Partial.begin(), Partial.end(), "") > 0) {
508  detail::write<binary>::op<Opts>(value, ctx, buffer, ix);
509  }
510  else {
511  using V = std::decay_t<T>;
512  static_assert(detail::glaze_object_t<V> || detail::writable_map_t<V>,
513  "Only object types are supported for partial.");
514  static constexpr auto sorted = sort_json_ptrs(Partial);
515  static constexpr auto groups = glz::group_json_ptrs<sorted>();
516  static constexpr auto N = std::tuple_size_v<std::decay_t<decltype(groups)>>;
517 
518  constexpr uint8_t type = 0; // string
519  constexpr uint8_t tag = tag::object | type;
520  detail::dump_type(tag, buffer, ix);
521 
522  detail::dump_compressed_int<N>(buffer, ix);
523 
524  if constexpr (detail::glaze_object_t<std::decay_t<T>>) {
525  for_each<N>([&](auto I) {
526  static constexpr auto group = glz::tuplet::get<I>(groups);
527 
528  static constexpr auto key = std::get<0>(group);
529  static constexpr auto sub_partial = std::get<1>(group);
530  static constexpr auto frozen_map = detail::make_map<T>();
531  static constexpr auto member_it = frozen_map.find(key);
532  static_assert(member_it != frozen_map.end(), "Invalid key passed to partial write");
533  static constexpr auto index = member_it->second.index();
534  static constexpr decltype(auto) member_ptr = std::get<index>(member_it->second);
535 
536  detail::write<binary>::no_header<Opts>(key, ctx, buffer, ix);
537  std::ignore = write<sub_partial, Opts>(glz::detail::get_member(value, member_ptr), ctx, buffer, ix);
538  });
539  }
540  else if constexpr (detail::writable_map_t<std::decay_t<T>>) {
541  for_each<N>([&](auto I) {
542  static constexpr auto group = glz::tuplet::get<I>(groups);
543 
544  static constexpr auto key_value = std::get<0>(group);
545  static constexpr auto sub_partial = std::get<1>(group);
546  if constexpr (findable<std::decay_t<T>, decltype(key_value)>) {
547  detail::write<binary>::no_header<Opts>(key_value, ctx, buffer, ix);
548  auto it = value.find(key_value);
549  if (it != value.end()) {
550  std::ignore = write<sub_partial, Opts>(it->second, ctx, buffer, ix);
551  }
552  else {
553  we.ec = error_code::invalid_partial_key;
554  }
555  }
556  else {
557  static thread_local auto key =
558  typename std::decay_t<T>::key_type(key_value); // TODO handle numeric keys
559  detail::write<binary>::no_header<Opts>(key, ctx, buffer, ix);
560  auto it = value.find(key);
561  if (it != value.end()) {
562  std::ignore = write<sub_partial, Opts>(it->second, ctx, buffer, ix);
563  }
564  else {
565  we.ec = error_code::invalid_partial_key;
566  }
567  }
568  });
569  }
570  }
571 
572  return we;
573  }

575  template <auto& Partial, opts Opts, class T, output_buffer Buffer>
576  [[nodiscard]] inline write_error write(T&& value, Buffer& buffer) noexcept
577  {
578  if constexpr (detail::resizeable<Buffer>) {
579  if (buffer.empty()) {
580  buffer.resize(128);
581  }
582  }
583  context ctx{};
584  size_t ix = 0;
585  const auto error = write<Partial, Opts>(std::forward<T>(value), ctx, buffer, ix);
586  if constexpr (detail::resizeable<Buffer>) {
587  buffer.resize(ix);
588  }
589  return error;
590  }
