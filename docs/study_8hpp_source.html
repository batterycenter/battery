101  [[nodiscard]] expected<State, error_code> generate(const size_t i) noexcept
102  {
103  size_t deconst_index = i;
104  for (auto& param_set : param_sets) {
105  const auto this_size = (std::max)(param_set.elements.size(), size_t{1});
106  const auto this_index = deconst_index % this_size;
107  deconst_index /= this_size;
108  const auto ec = std::visit(
109  [&](auto&& param_ptr) {
110  using V = std::remove_pointer_t<std::decay_t<decltype(param_ptr)>>;
111 
112  auto& element = param_set.elements[this_index];
113  if (std::holds_alternative<double>(element)) {
114  if constexpr (std::is_convertible_v<V, double>) {
115  *param_ptr = static_cast<V>(std::get<double>(element));
116  }
117  else {
118  return error_code::elements_not_convertible_to_design;
119  }
120  }
121  else {
122  *param_ptr = std::get<V>(param_set.elements[this_index]);
123  }
124 
125  return error_code::none;
126  },
127  param_set.param_ptr);
128 
129  if (bool(ec)) {
130  return unexpected(ec);
131  }
132  }
133 
134  return state;
135  }

139  expected<param_set, parse_error> param_set_from_dist(const param& dist)
140  {
141  param_set param_set;
142 
143  parse_error pe{};
144  bool found = detail::seek_impl(
145  [&](auto&& val) {
146  if constexpr (std::is_assignable_v<basic, std::decay_t<decltype(val)>>) {
147  param_set.param_ptr = &val;
148  }
149  else {
150  pe.ec = error_code::no_matching_variant_type; // Study params only support basic types like
151  // double, int, bool, or std::string
152  }
153  },
154  state, dist.ptr);
155  if (!found) {
156  pe.ec = error_code::get_nonexistent_json_ptr;
157  return unexpected(pe);
158  }
159 
160  if (pe) {
161  return unexpected(pe);
162  }
163 
164  if (dist.distribution == "elements") {
165  std::visit(
166  [&](auto&& param_ptr) {
167  for (auto&& j : dist.range) {
168  auto elem = *param_ptr;
169  pe = read_json(elem, j.str);
170  if (pe) {
171  return;
172  }
173  param_set.elements.emplace_back(elem);
174  }
175  },
176  param_set.param_ptr);
177  if (pe) {
178  return unexpected(pe);
179  }
180  }
181  else if (dist.distribution == "linspace") {
182  if (dist.range.size() != 3) {
183  return unexpected(parse_error{
184  error_code::invalid_distribution_elements}); // distribution's range must have 3 elements
185  }
186 
187  double start{};
188  double step{};
189  double stop{};
190 
191  if (auto ec = read_json(start, dist.range[0].str)) {
192  return unexpected(ec);
193  }
194  if (auto ec = read_json(step, dist.range[1].str)) {
195  return unexpected(ec);
196  }
197  if (auto ec = read_json(stop, dist.range[2].str)) {
198  return unexpected(ec);
199  }
200 
201  if (start > stop) {
202  std::swap(start, stop);
203  }
204 
205  for (; start <= stop; start += step) {
206  param_set.elements.emplace_back(start);
207  }
208  }
209  else {
210  return unexpected(parse_error{error_code::unknown_distribution});
211  }
212 
213  return param_set;
214  }

316  void resample(double ratio)
317  {
318  std::shuffle(std::begin(resample_indices), std::end(resample_indices), engine);
319  std::size_t to_resample = static_cast<std::size_t>(std::ceil(ratio * params_per_state.size()));
320 
321  for (std::size_t i = 0; i < to_resample; ++i) {

332  {

334 
335  parse_error pe{};
336  bool found = detail::seek_impl(
337  [&](auto&& val) {
338  if constexpr (std::is_assignable_v<basic, std::decay_t<decltype(val)>>) {
339  result.param_ptr = &val;
340  }
341  else {
342  pe.ec = error_code::no_matching_variant_type; // Study params only support basic types like
343  // double, int, bool, or std::string
344  }
345  },
346  state, dist.ptr);
347  if (!found) {
348  pe.ec = error_code::get_nonexistent_json_ptr;
349  return unexpected(pe);
350  }
351 
352  if (pe) {
353  return unexpected(pe);
354  }
355 
356  if (dist.distribution == "elements") {
357  if (dist.range.size() == 0) {
358  return unexpected(parse_error{error_code::invalid_distribution_elements});
359  }
360  std::vector<basic> elements{};
361  std::visit(
362  [&](auto&& param_ptr) {
363  for (auto&& json : dist.range) {
364  auto elem = *param_ptr;
365  read_json(elem, json.str);
366  elements.emplace_back(elem);
367  }
368  },
369  result.param_ptr);
370  result.gen = [this, dist = std::uniform_int_distribution<std::size_t>(0, dist.range.size() - 1),
371  elements = std::move(elements)]() mutable {
372  std::size_t element_index = dist(this->engine);
373  return elements[element_index];
374  };
375  }
376  else if (dist.distribution == "linspace") {
377  if (dist.range.size() != 3) {
378  return unexpected(parse_error{
379  error_code::invalid_distribution_elements}); // distribution's range must have 3 elements
380  }
381 
382  double start{};
383  double stop{};
384 
385  if (auto ec = read_json(start, dist.range[0].str)) {
386  return unexpected(ec);
387  }
388  if (auto ec = read_json(stop, dist.range[2].str)) {
389  return unexpected(ec);
390  }
391 
392  if (start > stop) {
393  std::swap(start, stop);
394  }
395 
396  result.gen = [this, dist = std::uniform_real_distribution<double>(start, stop)]() mutable {
397  return dist(this->engine);
398  };
399  }
400  else if (dist.distribution == "uniform") {
401  if (dist.range.size() != 2) {
402  return unexpected(parse_error{
403  error_code::invalid_distribution_elements}); // distribution's range must have 3 elements
404  }
405 
406  double start{};
407  double stop{};
408 
409  if (auto ec = read_json(start, dist.range[0].str)) {
410  return unexpected(ec);
411  }
412  if (auto ec = read_json(stop, dist.range[1].str)) {
413  return unexpected(ec);
414  }
415 
416  if (start > stop) {
417  std::swap(start, stop);
418  }
419
