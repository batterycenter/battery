114  // In Binary mode this function is cheap, it reads the filesize from the filesystem.
115  // In Text mode this function is expensive, it reads the entire file into memory, just for the filesize.
116  // There is no other way to get the filesize in text mode, because the representation on-disk is different
117  // from what is read into memory. (Line endings)
118  std::streamsize file_size() const {
119  if (binary) { // Binary mode
120  return std::filesystem::file_size(convert(path));
121  }
122  else { // Text mode: read the entire file into memory to compensate line-endings
123  std::string temp; // Buffer is as large as the file on-disk -> Reading will make it smaller, but not larger
124  auto filesize_temp = std::filesystem::file_size(convert(path));
125  temp.resize(filesize_temp);
126 
127  // Read file
128  fs::ifstream file(path, Mode::TEXT);
129  file.read(temp.data(), temp.size());
130  return static_cast<size_t>(file.gcount());
131  }
132  }
