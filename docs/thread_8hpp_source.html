61  explicit thread(Fn&& func, Args&&... args)
62  : std::jthread([func = std::forward<Fn>(func), ...args = std::forward<Args>(args), this]() mutable {
63  m_waitRunningPromise.set_value();
64  if constexpr (std::is_invocable_v<std::decay_t<Fn>, std::stop_token, std::decay_t<Args>...>) {
65  if (!catch_common_exceptions(std::move(func), get_stop_token(), std::move(args)...)) {
66  if (b::ConsoleApplication::instanceExists()) {
67  b::ConsoleApplication::get().stopApplication();
68  }
69  }
70  } else {

73  b::ConsoleApplication::get().stopApplication();
74  }
75  }
76  }

78  m_waitRunningPromise.get_future().get();

96  template<class... Args>
97  inline static bool catch_common_exceptions(Args&&... args) {
98  if (!m_catchCommonExceptionsEnabled) { // Catching is disabled: Call and return
99  std::invoke(std::forward<Args>(args)...);
100  return true;
101  }
102 
103  try { // Otherwise catch exceptions properly
104  std::invoke(std::forward<Args>(args)...);
105  return true;
106  }
107  catch (const std::bad_expected_access<std::error_code>& err) {
108  auto str = b::format("Unhandled exception: Bad expected access of <std::error_code>: {}",
109  err.error().message());
110  b::log::core::critical(str);
111  if (m_messageBoxOnExceptionEnabled) {
112  b::message_box_error(str);
113  }
114  }
115  catch (const std::exception& err) {
116  auto str = b::format("Unhandled [std::exception]: {}", err.what());
117  b::log::core::critical(str);
118  if (m_messageBoxOnExceptionEnabled) {
119  b::message_box_error(str);
120  }
121  }
122  catch (...) {
123  auto str = b::format("Unhandled exception: unknown type, no further information");
124  b::log::core::critical(str);
125  if (m_messageBoxOnExceptionEnabled) {
126  b::message_box_error(str);
127  }
128  }
129  return false;
130  }
