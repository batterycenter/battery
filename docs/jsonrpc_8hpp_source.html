278  using raw_call_return_t = std::vector<raw_response_t>;
279  // Return json stringified std::vector<response_t> meaning it can both be error and non-error and batch request
280  // response. If `id` is null in json_request a response will not be generated
281  // If you would like to handle errors for each request:
282  // auto response_vector = server.call<decltype(server_instance)::raw_call_return_t>("...");
283  // std::string response = glz::write_json(response_vector);
284  template <concepts::call_return_type return_t = std::string>
285  return_t call(std::string_view json_request)
286  {
287  constexpr auto return_helper = []<typename input_type>(input_type && response) -> auto
288  {
289  if constexpr (std::same_as<return_t, std::string>) {
290  return glz::write_json(std::forward<input_type>(response));
291  }
292  else if constexpr (std::same_as<input_type, raw_response_t>) {
293  return std::vector<raw_response_t>{std::forward<input_type>(response)};
294  }
295  else {
296  return std::forward<input_type>(response);
297  }
298  };
299 
300  if (auto parse_err{glz::validate_json(json_request)}) {
301  return return_helper(
302  raw_response_t{rpc::error{error_e::parse_error, format_error(parse_err, json_request)}});
303  }
304 
305  auto batch_requests{glz::read_json<std::vector<glz::raw_json_view>>(json_request)};
306  if (batch_requests.has_value() && batch_requests.value().empty()) {
307  return return_helper(raw_response_t{rpc::error{error_e::invalid_request}});
308  }
309  if (batch_requests.has_value()) {
310  return return_helper(batch_request(batch_requests.value()));
311  }
312 
313  // At this point we are with a single request
314  auto response{per_request(json_request)};
315  if (response.has_value()) {
316  return return_helper(std::move(response.value()));
317  }
318  return {};
319  }

321  private:
322  auto per_request(std::string_view json_request) -> std::optional<response_t<glz::raw_json>>
323  {
324  expected<generic_request_t, glz::parse_error> request{glz::read_json<generic_request_t>(json_request)};
325 
326  if (!request.has_value()) {
327  // Failed, but let's try to extract the `id`
328  auto id{glz::get_as_json<id_t, "/id">(json_request)};
329  if (!id.has_value()) {
330  return raw_response_t{rpc::error::invalid(request.error(), json_request)};
331  }
332  return raw_response_t{std::move(id.value()), rpc::error::invalid(request.error(), json_request)};
333  }
334 
335  auto& req{request.value()};
336  if (req.version != rpc::supported_version) {
337  return raw_response_t{std::move(req.id), rpc::error::version(req.version)};
338  }
339 
340  std::optional<response_t<glz::raw_json>> return_v{};
341  bool method_found = methods.any([&json_request, &req, &return_v](auto&& method) -> bool {
342  using meth_t = std::remove_reference_t<decltype(method)>;
343  if (req.method != meth_t::name_v) {
344  return false;
345  }
346  auto const& params_request{glz::read_json<typename meth_t::request_t>(json_request)};
347  if (params_request.has_value()) {
348  expected<typename meth_t::result_t, rpc::error> result{
349  std::invoke(method.callback, params_request->params)};
350  if (result.has_value()) {
351  return_v = raw_response_t{std::move(req.id), glz::write_json(result)};
352  if (std::holds_alternative<glz::json_t::null_t>(req.id)) {
353  // rpc notification requires no response
354  return_v = std::nullopt;
355  }
356  }
357  else {
358  return_v = raw_response_t{std::move(req.id), std::move(result.error())};
359  }
360  }
361  else {
362  // TODO: Finish thought, glaze does not error on missing field:
363  // https://github.com/stephenberry/glaze/issues/207
364  return_v = raw_response_t{std::move(req.id), rpc::error::invalid(params_request.error(), json_request)};
365  }
366  return true;
367  });
368  if (!method_found) {
369  return raw_response_t{std::move(req.id), rpc::error::method(req.method)};
370  }
371  return return_v;
372  };

374  auto batch_request(const std::vector<glz::raw_json_view>& batch_requests)
375  {
376  std::vector<response_t<glz::raw_json>> return_vec;
377  return_vec.reserve(batch_requests.size());
378  for (auto&& request : batch_requests) {
379  auto response{per_request(request.str)};
380  if (response.has_value()) {
381  return_vec.emplace_back(std::move(response.value()));
382  }
383  }
384  return return_vec;

393  rpc::error call(std::string_view json_response)
394  {
395  expected<generic_response_t, glz::parse_error> response{glz::read_json<generic_response_t>(json_response)};
396 
397  if (!response.has_value()) {
398  return rpc::error{rpc::error_e::parse_error, format_error(response.error(), json_response)};
399  }
400 
401  auto& res{response.value()};
402 
403  rpc::error return_v{};
404  bool id_found = methods.any([&json_response, &res, &return_v](auto&& method) -> bool {
405  using meth_t = std::remove_reference_t<decltype(method)>;
406 
407  if (auto it{method.pending_requests.find(res.id)}; it != std::end(method.pending_requests)) {
408  auto [id, callback] = *it;
409 
410  expected<typename meth_t::response_t, glz::parse_error> response{
411  glz::read_json<typename meth_t::response_t>(json_response)};
412 
413  if (!response.has_value()) {
414  return_v = rpc::error{rpc::error_e::parse_error, format_error(response.error(), json_response)};
415  }
416  else {
417  auto& res_typed{response.value()};
418  if (res_typed.result.has_value()) {
419  std::invoke(callback, res_typed.result.value(), res_typed.id);
420  }
421  else if (res_typed.error.has_value()) {
422  std::invoke(callback, unexpected(res_typed.error.value()), res_typed.id);
423  }
424  else {
425  return_v = rpc::error{rpc::error_e::parse_error, R"(Missing key "result" or "error" in response)"};
426  }
427  }
428  method.pending_requests.erase(it);
429  return true;
430  }
431  return false;
432  });
433 
434  if (!id_found) [[unlikely]] {
435  if (!return_v) {
436  if (std::holds_alternative<std::string_view>(res.id)) {
437  return_v = rpc::error{error_e::internal,
438  "id: '" + std::string(std::get<std::string_view>(res.id)) + "' not found"};
439  }
440  else {
441  return_v = rpc::error{error_e::internal, "id: " + glz::write_json(res.id) + " not found"};
442  }
443  }
444  }
445 
446  return return_v;
447  }

449  // callback should be of type std::function<void(glz::expected<result_t, rpc::error> const&, jsonrpc_id_type
450  // const&)> where result_t is the result type declared for the given method name returns the request string and
451  // whether the callback was inserted into the queue if the callback was not inserted into the queue it can mean
452  // that the input was a notification or more serious, conflicting id, the provided id should be unique!
453  template <basic_fixed_string method_name>
454  [[nodiscard]] std::pair<std::string, bool> request(id_t&& id, auto&& params, auto&& callback)
455  {
456  constexpr bool method_found = ((method_type::name_v == method_name) || ...);
457  static_assert(method_found, "Method not declared in client.");
458 
459  using params_type = std::remove_cvref_t<decltype(params)>;
460  constexpr bool params_type_found = (std::is_same_v<typename method_type::params_t, params_type> || ...);
461  static_assert(params_type_found, "Input parameters do not match constructed client parameter types.");
462 
463  constexpr bool method_params_match =
464  ((method_type::name_v == method_name && std::is_same_v<typename method_type::params_t, params_type>) ||
465  ...);
466  static_assert(method_params_match, "Method name and given params type do not match.");
467 
468  rpc::request_t req{std::forward<decltype(id)>(id), method_name.view(), std::forward<decltype(params)>(params)};
469 
470  if (std::holds_alternative<glz::json_t::null_t>(id)) {
471  return {glz::write_json(std::move(req)), false};
472  }
473 
474  // Some bookkeeping store the callback
475  bool inserted{false};
476  methods.any([&inserted, &req, cb = std::forward<decltype(callback)>(callback)](auto&& method) -> bool {
477  using meth_t = std::remove_reference_t<decltype(method)>;
478  if constexpr (method_name == meth_t::name_v) {
479  [[maybe_unused]] decltype(method.pending_requests.begin()) unused{}; // iterator_t
480  std::tie(unused, inserted) = method.pending_requests.emplace(std::make_pair(req.id, cb));
481  return true; // break methods loop
482  }
483  else {
